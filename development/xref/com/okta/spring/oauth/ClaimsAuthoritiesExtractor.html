16  package com.okta.spring.oauth;
17  
18  import org.springframework.boot.autoconfigure.security.oauth2.resource.AuthoritiesExtractor;
19  import org.springframework.security.core.GrantedAuthority;
20  import org.springframework.security.core.authority.SimpleGrantedAuthority;
21  
22  import java.util.Collections;
23  import java.util.List;
24  import java.util.Map;
25  import java.util.stream.Collectors;
26  
27  /**
28   * A {@link AuthoritiesExtractor} converts a list of Groups/Roles in an JWT token into Spring Security
29   * {@link GrantedAuthority}.
30   * @since 0.2.0
31   */
32  public class ClaimsAuthoritiesExtractor implements AuthoritiesExtractor {
33  
34      private final String rolesClaimKey;
35  
36      public ClaimsAuthoritiesExtractor(String rolesClaimKey) {
37          this.rolesClaimKey = rolesClaimKey;
38      }
39  
40      @Override
41      public List<GrantedAuthority> extractAuthorities(Map<String, Object> map) {
42          // extract the string groups from map
43          List groups= Collections.emptyList();
44          if (map.containsKey(rolesClaimKey)) {
45              Object rawGroups = map.get(rolesClaimKey);
46              if (rawGroups instanceof List) {
47                  groups = (List) rawGroups;
48              }
49          }
50  
51          // convert them to authorities
52          return (List<GrantedAuthority>) groups.stream()
53                  .filter(String.class::isInstance)
54                  .map(group -> new SimpleGrantedAuthority(group.toString()))
55                  .collect(Collectors.toList());
56      }
57  }
