16  package com.okta.spring.boot.oauth;
17  
18  import com.okta.spring.boot.oauth.config.OktaOAuth2Properties;
19  import org.springframework.beans.factory.annotation.Qualifier;
20  import org.springframework.boot.autoconfigure.AutoConfigureBefore;
21  import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
22  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
23  import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
24  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication;
25  import org.springframework.boot.context.properties.EnableConfigurationProperties;
26  import org.springframework.context.annotation.Bean;
27  import org.springframework.context.annotation.Configuration;
28  import org.springframework.context.annotation.Import;
29  import org.springframework.security.config.annotation.web.reactive.EnableWebFluxSecurity;
30  import org.springframework.security.config.web.server.ServerHttpSecurity;
31  import org.springframework.security.oauth2.client.oidc.userinfo.OidcReactiveOAuth2UserService;
32  import org.springframework.security.oauth2.client.registration.ClientRegistration;
33  import org.springframework.security.oauth2.client.userinfo.OAuth2UserRequest;
34  import org.springframework.security.oauth2.client.userinfo.ReactiveOAuth2UserService;
35  import org.springframework.security.oauth2.core.user.OAuth2User;
36  import org.springframework.security.oauth2.jwt.ReactiveJwtDecoder;
37  import org.springframework.security.web.server.SecurityWebFilterChain;
38  import reactor.core.publisher.Flux;
39  
40  import java.util.Collection;
41  
42  @Configuration
43  @AutoConfigureBefore(name = {
44      "org.springframework.boot.autoconfigure.security.oauth2.client.reactive.ReactiveOAuth2ClientAutoConfiguration",
45      "org.springframework.boot.autoconfigure.security.oauth2.resource.reactive.ReactiveOAuth2ResourceServerAutoConfiguration"})
46  @EnableConfigurationProperties(OktaOAuth2Properties.class)
47  @ConditionalOnOktaClientProperties
48  @ConditionalOnWebApplication(type = ConditionalOnWebApplication.Type.REACTIVE)
49  @ConditionalOnClass({ Flux.class, EnableWebFluxSecurity.class, ClientRegistration.class })
50  @Import(AuthorityProvidersConfig.class)
51  class ReactiveOktaOAuth2AutoConfig {
52  
53      @Bean
54      @ConditionalOnMissingBean
55      ReactiveOAuth2UserService<OAuth2UserRequest, OAuth2User> oauth2UserService(Collection<AuthoritiesProvider> authoritiesProviders) {
56          return new ReactiveOktaOAuth2UserService(authoritiesProviders);
57      }
58  
59      @Bean
60      @ConditionalOnMissingBean
61      OidcReactiveOAuth2UserService oidcUserService(Collection<AuthoritiesProvider> authoritiesProviders,
62                                                    @Qualifier("oauth2UserService") ReactiveOAuth2UserService<OAuth2UserRequest, OAuth2User> oAuth2UserService) {
63          return new ReactiveOktaOidcUserService(authoritiesProviders, oAuth2UserService);
64      }
65  
66      @Bean
67      @ConditionalOnBean(ReactiveJwtDecoder.class)
68      @ConditionalOnMissingBean(SecurityWebFilterChain.class)
69      SecurityWebFilterChain springSecurityFilterChain(ServerHttpSecurity http, ReactiveJwtDecoder jwtDecoder) {
70          // as of Spring Security 5.4 the default chain uses oauth2Login OR a JWT resource server (NOT both)
71          // this does the same as both defaults merged together (and provides the previous behavior)
72          http.authorizeExchange().anyExchange().authenticated();
73          http.oauth2Login();
74          http.oauth2Client();
75          http.oauth2ResourceServer((server) -> customDecoder(server, jwtDecoder));
76          return http.build();
77      }
78  
79      private void customDecoder(ServerHttpSecurity.OAuth2ResourceServerSpec server, ReactiveJwtDecoder decoder) {
80          server.jwt((jwt) -> jwt.jwtDecoder(decoder));
81      }
82  }
