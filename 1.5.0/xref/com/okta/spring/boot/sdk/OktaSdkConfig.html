43  /**
44   * Configure Okta's management SDK, and expose it as a Bean.
45   *
46   * @since 0.3.0
47   */
48  @Configuration
49  @Conditional(OktaSdkConfig.OktaApiTokenCondition.class)
50  @ConditionalOnClass(Client.class)
51  @EnableConfigurationProperties(OktaClientProperties.class)
52  public class OktaSdkConfig {
53  
54      private final OktaClientProperties oktaClientProperties;
55      private final org.springframework.cache.CacheManager springCacheManager;
56  
57      public OktaSdkConfig(OktaClientProperties oktaClientProperties,
58                           @Autowired(required = false) org.springframework.cache.CacheManager springCacheManager) {
59  
60          this.oktaClientProperties = oktaClientProperties;
61          this.springCacheManager = springCacheManager;
62      }
63  
64      @Bean
65      protected Client oktaSdkClient() {
66          ClientBuilder builder = Clients.builder()
67                  .setCacheManager(oktaSdkCacheManager())
68                  .setAuthenticationScheme(oktaSdkAuthenticationScheme())
69                  .setConnectionTimeout(oktaClientProperties.getConnectionTimeout())
70                  .setClientCredentials(oktaSdkClientCredentials())
71                  .setOrgUrl(oktaClientProperties.getOrgUrl());
72  
73          Proxy proxy = oktaSdkProxy();
74          if (proxy != null) {
75              builder.setProxy(oktaSdkProxy());
76          }
77  
78          return builder.build();
79      }
80  
81      private AuthenticationScheme oktaSdkAuthenticationScheme() {
82          return AuthenticationScheme.SSWS;
83      }
84  
85      private Proxy oktaSdkProxy() {
86  
87          OktaClientProperties.ClientProxyInfo proxyInfo = oktaClientProperties.getProxy();
88          if (proxyInfo == null || !Strings.hasText(proxyInfo.getHostname())) {
89              return null;
90          }
91  
92          Proxy proxy;
93  
94          if (Strings.hasText(proxyInfo.getUsername()) || Strings.hasText(proxyInfo.getPassword())) {
95              proxy = new Proxy(proxyInfo.getHostname(), proxyInfo.getPort(), proxyInfo.getUsername(), proxyInfo.getPassword());
96          } else {
97              proxy = new Proxy(proxyInfo.getHostname(), proxyInfo.getPort());
98          }
99  
100         return proxy;
101     }
102 
103     private CacheManager oktaSdkCacheManager() {
104         return (springCacheManager != null) ?
105              new SpringCacheManager(springCacheManager) : null;
106     }
107 
108     @Bean
109     @ConditionalOnMissingBean
110     protected ClientCredentials oktaSdkClientCredentials() {
111         return new TokenClientCredentials(oktaClientProperties.getToken());
112     }
113 
114     /**
115      * Spring Boot conditional based on the existance of the {code}okta.client.token{code} property.
116      */
117     static class OktaApiTokenCondition extends SpringBootCondition {
118 
119         @Override
120         public ConditionOutcome getMatchOutcome(ConditionContext context, AnnotatedTypeMetadata metadata) {
121 
122             ConditionMessage.Builder message = ConditionMessage.forCondition("Okta Api Token Condition");
123             String tokenValue = context.getEnvironment().getProperty("okta.client.token");
124             if (StringUtils.hasText(tokenValue)) {
125                 return ConditionOutcome.match(message.foundExactly("provided API token"));
126             }
127             return ConditionOutcome.noMatch(message.didNotFind("provided API token").atAll());
128         }
129     }
130 }
