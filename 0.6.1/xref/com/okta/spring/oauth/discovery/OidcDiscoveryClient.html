16  package com.okta.spring.oauth.discovery;
17  
18  import com.okta.spring.oauth.http.UserAgent;
19  import org.springframework.http.HttpHeaders;
20  import org.springframework.http.HttpRequest;
21  import org.springframework.http.client.ClientHttpRequestExecution;
22  import org.springframework.http.client.ClientHttpRequestInterceptor;
23  import org.springframework.http.client.ClientHttpResponse;
24  import org.springframework.util.Assert;
25  import org.springframework.web.client.RestTemplate;
26  import org.springframework.web.util.UriComponentsBuilder;
27  
28  import java.io.IOException;
29  import java.net.URI;
30  
31  /**
32   * OIDC discovery client.
33   * NOTE: parts of this class were heavily borrowed from {code}org.springframework.security.oauth2.client.discovery.ProviderDiscoveryClient{code}
34   * @since 0.2.0
35   */
36  public class OidcDiscoveryClient {
37  
38      private final URI issuerUri;
39      private final RestTemplate restTemplate = restTemplate();
40  
41      public OidcDiscoveryClient(String issuer) {
42  
43          Assert.hasText(issuer, "issuer cannot be empty");
44          try {
45              this.issuerUri = UriComponentsBuilder.fromHttpUrl(issuer)
46                      .path("/.well-known/openid-configuration")
47                      .build()
48                      .encode()
49                      .toUri();
50          } catch (Exception ex) {
51              throw new IllegalArgumentException("Invalid URI for issuer: " + ex.getMessage(), ex);
52          }
53      }
54  
55      public OidcDiscoveryMetadata discover() {
56          return this.restTemplate.getForObject(issuerUri, OidcDiscoveryMetadata.class);
57      }
58  
59      private static RestTemplate restTemplate() {
60          RestTemplate template = new RestTemplate();
61          template.getInterceptors().add(new UserAgentInterceptor());
62          return template;
63      }
64  
65      private static class UserAgentInterceptor implements ClientHttpRequestInterceptor {
66          @Override
67          public ClientHttpResponse intercept(HttpRequest request, byte[] body, ClientHttpRequestExecution execution) throws IOException {
68              request.getHeaders().add(HttpHeaders.USER_AGENT, UserAgent.getUserAgentString());
69              return execution.execute(request, body);
70          }
71      }
72  }
