34  /**
35   * This class is in charge of constructing the <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.43">
36   * User-Agent http header</a> string that will be sent to Okta in order to describe the current running environment of this Java SDK.
37   * <p>
38   * The form of this string is the concatenation of the following sub-items:
39   * <ol>
40   *     <li>The okta integration and version separated by a '/'.  If there is no integration being used, this can be omitted
41   *     <li>The okta sdk and version separated by a '/'
42   *     <li>The runtime information (runtime/version)
43   *     <ol type="a">
44   *          <li>Integration Runtime (if there is no integration being used, this can be omitted)
45   *          <li>SDK Runtime
46   *     </ol>
47   *     <li>The OS common name and version separated by a '/'.
48   *     <li>All other system information included in parentheses
49   * </ol>
50   * <p>
51   * The User-Agent value is created when this class is loaded. The string can be obtained just by invoking
52   * {@link UserAgent#getUserAgentString() UserAgent.getUserAgentString()}.
53   * <p>
54   * This is a sample User-Agent string:
55   * <i>okta-spring-security/0.7.0 okta-sdk-java/0.5.0 spring/4.0.4.RELEASE java/1.7.0_45 Mac OS X/10.9.2 (spring-security/3.2.0.RELEASE jetty/8.1.5.v20120716)</i>
56   *
57   * @since 0.4.0
58   */

137     private static String createUserAgentString() {
138         String userAgent =
139                 getOktaSpringString() +                 // okta-spring-security
140                 getOktaShiroString() +                  // okta-shiro
141                 getOktaSDKComponentsString() +          // okta-servlet-java | okta-spring-boot-starter
142                 getOktaSdkString() +                    // okta-sdk-java
143                 getSecurityFrameworkString() +          // shiro | spring-security
144                 getIntegrationRuntimeString() +         // spring
145                 getSpringBootString() +                 // spring-boot
146                 getWebServerString() +                  // tomcat | jetty | jboss | websphere | glassfish | weblogic | wildfly
147                 getJavaSDKRuntimeString() +             // java
148                 getOSString();                          // Mac OS X
149         return userAgent.trim();
150     }

232     private static String getWebServerString() {
233         String webServerString;
234         //Glassfish uses Tomcat internally, therefore the Glassfish check must be carried out before Tomcat's
235         webServerString = getFullEntryStringUsingManifest(WEB_SERVER_GLASSFISH_CLASS, WEB_SERVER_GLASSFISH_ID);
236         if (StringUtils.hasText(webServerString)) {
237             return webServerString;
238         }
239         webServerString = getFullEntryStringUsingManifest(WEB_SERVER_TOMCAT_BOOTSTRAP_CLASS, WEB_SERVER_TOMCAT_ID);
240         if (StringUtils.hasText(webServerString)) {
241             return webServerString;
242         }
243         webServerString = getFullEntryStringUsingManifest(WEB_SERVER_TOMCAT_EMBEDDED_CLASS, WEB_SERVER_TOMCAT_ID);
244         if (StringUtils.hasText(webServerString)) {
245             return webServerString;
246         }
247         webServerString = getFullEntryStringUsingManifest(WEB_SERVER_JETTY_CLASS, WEB_SERVER_JETTY_ID);
248         if (StringUtils.hasText(webServerString)) {
249             return webServerString;
250         }
251         //WildFly must be before JBoss
252         webServerString = getWildFlyEntryString();
253         if (StringUtils.hasText(webServerString)) {
254             return webServerString;
255         }
256         webServerString = getFullEntryStringUsingManifest(WEB_SERVER_JBOSS_CLASS, WEB_SERVER_JBOSS_ID);
257         if (StringUtils.hasText(webServerString)) {
258             return webServerString;
259         }
260         webServerString = getWebSphereEntryString();
261         if (StringUtils.hasText(webServerString)) {
262             return webServerString;
263         }
264         webServerString = getWebLogicEntryString();
265         if (StringUtils.hasText(webServerString)) {
266             return webServerString;
267         }
268 
269         return "";
270     }

313                         return WEB_SERVER_WILDFLY_ID + VERSION_SEPARATOR + wildFlyPkg.getImplementationVersion() + ENTRY_SEPARATOR;

326     /**
327      * WARNING: This method must never be invoked unless we already know that the class identified by the parameter {@code fqcn}
328      * really exists in the classpath. For example, we first need to assure that {@code Classes.isAvailable(fqcn))} is <code>TRUE</code>
329      */
330     private static String getVersionInfoFromPomProperties(String fqcn) {
331         String version = "unknown";
332         try{
333             Class clazz = ClassUtils.forName(fqcn, null);
334             String className = clazz.getSimpleName() + ".class";
335             String classPath = clazz.getResource(className).toString();
336 
337             String jarPath = null;
338             if (classPath.startsWith("jar:file:")) {
339                 //Let's remove "jar:file:" from the beginning and also the className
340                 jarPath = classPath.subSequence(9, classPath.lastIndexOf("!")).toString();
341             } else if (classPath.startsWith("vfs:")) {
342                 //Let's remove "vfs:" from the beginning and also the className
343                 jarPath = classPath.subSequence(4, classPath.lastIndexOf(".jar") + 4).toString();
344             }
345 
346             if (jarPath == null) {
347                 //we were not able to obtain the jar path. Let's abort
348                 return version;
349             }
350 
351             Enumeration<JarEntry> enumeration;
352             String pomPropertiesPath;
353             try (JarFile jarFile = new JarFile(jarPath)) {
354                 enumeration = jarFile.entries();
355             }
356             pomPropertiesPath = null;
357             while (enumeration.hasMoreElements()) {
358                 JarEntry entry = enumeration.nextElement();
359                 if (entry.getName().endsWith("pom.properties")) {
360                     pomPropertiesPath = entry.getName();
361                     break;
362                 }
363             }
364             if (pomPropertiesPath != null) {
365                 try (BufferedReader reader = new BufferedReader(new InputStreamReader(clazz.getResourceAsStream("/" + pomPropertiesPath), StandardCharsets.UTF_8))) {
366                     String line;
367                     while ((line = reader.readLine()) != null) {
368                         if (line.startsWith("version=")) {
369                             version = line.split("=")[1];
370                             break;
371                         }
372                     }
373                 }
374             }
375         } catch (RuntimeException e) {
376             throw e;
377         } catch (Exception e) { //NOPMD
378             //Either the jar file or the internal "pom.properties" file could not be read, there is nothing we can do...
379         }
380         return version;
381     }

383     /**
384      * WARNING: This method must never be invoked unless we already know that the class identified by the parameter {@code fqcn}
385      * really exists in the classpath. For example, we first need to assure that {@code Classes.isAvailable(fqcn))} is <code>TRUE</code>
386      */
387     private static String getVersionInfoInManifest(String fqcn){
388         String version = null;
389         try {
390             //get class package
391             Package thePackage = ClassUtils.forName(fqcn, null).getPackage();
392             //examine the package object
393             version = thePackage.getSpecificationVersion();
394             if (!StringUtils.hasText(version)) {
395                 version = thePackage.getImplementationVersion();
396             }
397         } catch (ClassNotFoundException e) {
398             LOG.debug("Failed resolve version for class '{}'", fqcn, e);
399         }
400 
401         if (!StringUtils.hasText(version)) {
402             version = "null";
403         }
404         return version;
405     }

407     /**
408      * This method should only be invoked after already knowing that the class identified by {@code WEB_SERVER_WEBSPHERE_CLASS}
409      * really exists in the classpath. For example, it can be checked that {@code Classes.isAvailable(WEB_SERVER_WEBSPHERE_CLASS))}
410      * is {@code TRUE}
411      */
412     private static String getWebSphereVersion() {
413         try {
414             Class<?> versionClass = Class.forName(WEB_SERVER_WEBSPHERE_CLASS);
415             Object versionInfo = versionClass.newInstance();
416             Method method = versionClass.getDeclaredMethod("runReport", String.class, PrintWriter.class);
417             StringWriter stringWriter = new StringWriter();
418             PrintWriter printWriter = new PrintWriter(stringWriter);
419             method.invoke(versionInfo, "", printWriter);
420             String version = stringWriter.toString();
421             // version looks like this, so we need to "substring" it:
422             //
423             //
424             //IBM WebSphere Product Installation Status Report
425             //--------------------------------------------------------------------------------
426             //
427             //Report at date and time August 13, 2014 1:12:06 PM ART
428             //
429             //Installation
430             //--------------------------------------------------------------------------------
431             //Product Directory        C:\Program Files\IBM\WebSphere\AppServer
432             //Version Directory        C:\Program Files\IBM\WebSphere\AppServer\properties\version
433             //DTD Directory            C:\Program Files\IBM\WebSphere\AppServer\properties\version\dtd
434             //Log Directory            C:\Documents and Settings\All Users\Application Data\IBM\Installation Manager\logs
435             //
436             //Product List
437             //--------------------------------------------------------------------------------
438             //BASETRIAL                installed
439             //
440             //Installed Product
441             //--------------------------------------------------------------------------------
442             //Name                  IBM WebSphere Application Server
443             //Version               8.5.5.2
444             //ID                    BASETRIAL
445             //Build Level           cf021414.01
446             //Build Date            4/8/14
447             //Package               com.ibm.websphere.BASETRIAL.v85_8.5.5002.20140408_1947
448             //Architecture          x86 (32 bit)
449             //Installed Features    IBM 32-bit WebSphere SDK for Java
450             //WebSphere Application Server Full Profile
451 
452             version = version.substring(version.indexOf("Installed Product"));
453             version = version.substring(version.indexOf("Version"));
454             version = version.substring(version.indexOf(" "), version.indexOf("\n")).trim();
455             return version;
456 
457         } catch (RuntimeException e) {
458             throw e;
459         } catch (Exception e) { //NOPMD
460             //there was a problem obtaining the WebSphere version
461         }
462         //returning null so we can identify in the User-Agent String that we are not properly handling some WebSphere version
463         return "null";
464     }
