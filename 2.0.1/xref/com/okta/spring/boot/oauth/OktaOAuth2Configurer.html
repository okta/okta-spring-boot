16  package com.okta.spring.boot.oauth;
17  
18  import com.okta.spring.boot.oauth.config.OktaOAuth2Properties;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  import org.springframework.boot.autoconfigure.security.oauth2.client.OAuth2ClientProperties;
22  import org.springframework.context.ApplicationContext;
23  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
24  import org.springframework.security.config.annotation.web.configurers.AbstractHttpConfigurer;
25  import org.springframework.security.config.annotation.web.configurers.oauth2.server.resource.OAuth2ResourceServerConfigurer;
26  import org.springframework.security.oauth2.client.endpoint.DefaultAuthorizationCodeTokenResponseClient;
27  import org.springframework.security.oauth2.client.endpoint.OAuth2AccessTokenResponseClient;
28  import org.springframework.security.oauth2.client.endpoint.OAuth2AuthorizationCodeGrantRequest;
29  import org.springframework.security.oauth2.client.oidc.web.logout.OidcClientInitiatedLogoutSuccessHandler;
30  import org.springframework.web.client.RestTemplate;
31  
32  import java.lang.reflect.Field;
33  import java.security.AccessController;
34  import java.security.PrivilegedAction;
35  import java.util.Optional;
36  
37  import static org.springframework.util.StringUtils.isEmpty;
38  
39  final class OktaOAuth2Configurer extends AbstractHttpConfigurer<OktaOAuth2Configurer, HttpSecurity> {
40  
41      private static final Logger log = LoggerFactory.getLogger(OktaOAuth2Configurer.class);
42  
43      @SuppressWarnings("rawtypes")
44      @Override
45      public void init(HttpSecurity http) throws Exception {
46  
47          ApplicationContext context = http.getSharedObject(ApplicationContext.class);
48  
49          // make sure OktaOAuth2Properties are available
50          if (!context.getBeansOfType(OktaOAuth2Properties.class).isEmpty()) {
51              OktaOAuth2Properties oktaOAuth2Properties = context.getBean(OktaOAuth2Properties.class);
52  
53              // Auth Code Flow Config
54  
55              // if OAuth2ClientProperties bean is not available do NOT configure
56              if (!context.getBeansOfType(OAuth2ClientProperties.class).isEmpty()
57                  && !isEmpty(oktaOAuth2Properties.getIssuer())
58                  && !isEmpty(oktaOAuth2Properties.getClientId())) {
59                  // configure Okta user services
60                  configureLogin(http, oktaOAuth2Properties);
61  
62                  // check for RP-Initiated logout
63                  if (!context.getBeansOfType(OidcClientInitiatedLogoutSuccessHandler.class).isEmpty()) {
64                      http.logout().logoutSuccessHandler(context.getBean(OidcClientInitiatedLogoutSuccessHandler.class));
65                  }
66  
67                  // Resource Server Config
68  
69                  // if issuer is root org, use opaque token validation
70                  if (TokenUtil.isRootOrgIssuer(oktaOAuth2Properties.getIssuer())) {
71                      log.debug("Opaque Token validation/introspection will be configured.");
72                      configureResourceServerForOpaqueTokenValidation(http, oktaOAuth2Properties);
73                      return;
74                  }
75  
76                  OAuth2ResourceServerConfigurer oAuth2ResourceServerConfigurer = http.getConfigurer(OAuth2ResourceServerConfigurer.class);
77  
78                  if (getJwtConfigurer(oAuth2ResourceServerConfigurer).isPresent()) {
79                      log.debug("JWT configurer is set in OAuth resource server configuration. " +
80                          "JWT validation will be configured.");
81                      configureResourceServerForJwtValidation(http, oktaOAuth2Properties);
82                  } else if (getOpaqueTokenConfigurer(oAuth2ResourceServerConfigurer).isPresent()) {
83                      log.debug("Opaque Token configurer is set in OAuth resource server configuration. " +
84                          "Opaque Token validation/introspection will be configured.");
85                      configureResourceServerForOpaqueTokenValidation(http, oktaOAuth2Properties);
86                  } else {
87                      log.debug("Defaulting to Okta JWT resource server configuration.");
88                      configureResourceServerForJwtValidation(http, oktaOAuth2Properties);
89                  }
90              } else {
91                  log.debug("OAuth/OIDC Login not configured due to missing issuer, client-id, or client-secret property");
92              }
93          }
94      }
95  
96      private Optional<OAuth2ResourceServerConfigurer<?>.JwtConfigurer> getJwtConfigurer(OAuth2ResourceServerConfigurer<?> oAuth2ResourceServerConfigurer) throws IllegalAccessException {
97          if (oAuth2ResourceServerConfigurer != null) {
98              return getFieldValue(oAuth2ResourceServerConfigurer, "jwtConfigurer");
99          }
100         return Optional.empty();
101     }
102 
103     private Optional<OAuth2ResourceServerConfigurer<?>.OpaqueTokenConfigurer> getOpaqueTokenConfigurer(OAuth2ResourceServerConfigurer<?> oAuth2ResourceServerConfigurer) throws IllegalAccessException {
104         if (oAuth2ResourceServerConfigurer != null) {
105             return getFieldValue(oAuth2ResourceServerConfigurer, "opaqueTokenConfigurer");
106         }
107         return Optional.empty();
108     }
109 
110     private <T> Optional<T> getFieldValue(Object source, String fieldName) throws IllegalAccessException {
111         Field field = AccessController.doPrivileged((PrivilegedAction<Field>) () -> {
112             Field result = null;
113             try {
114                 result = OAuth2ResourceServerConfigurer.class.getDeclaredField(fieldName);
115                 result.setAccessible(true);
116             } catch (NoSuchFieldException e) {
117                 log.warn("Could not get field '" + fieldName + "' of {} via reflection",
118                     OAuth2ResourceServerConfigurer.class.getName(), e);
119             }
120             return result;
121         });
122 
123         if (field == null) {
124             String errMsg = "Expected field '" + fieldName + "' was not found in OAuth resource server configuration. " +
125                 "Version incompatibility with Spring Security detected." +
126                 "Check https://github.com/okta/okta-spring-boot for project updates.";
127             throw new RuntimeException(errMsg);
128         }
129 
130         return Optional.ofNullable((T) field.get(source));
131     }
132 
133     private void configureLogin(HttpSecurity http, OktaOAuth2Properties oktaOAuth2Properties) throws Exception {
134 
135         RestTemplate restTemplate = OktaOAuth2ResourceServerAutoConfig.restTemplate(oktaOAuth2Properties);
136 
137         http.oauth2Login()
138             .tokenEndpoint()
139             .accessTokenResponseClient(accessTokenResponseClient(restTemplate));
140 
141         if (oktaOAuth2Properties.getRedirectUri() != null) {
142             http.oauth2Login().redirectionEndpoint().baseUri(oktaOAuth2Properties.getRedirectUri());
143         }
144     }
145 
146     private void configureResourceServerForJwtValidation(HttpSecurity http, OktaOAuth2Properties oktaOAuth2Properties) throws Exception {
147         http.oauth2ResourceServer()
148             .jwt().jwtAuthenticationConverter(new OktaJwtAuthenticationConverter(oktaOAuth2Properties.getGroupsClaim()));
149     }
150 
151     private void configureResourceServerForOpaqueTokenValidation(HttpSecurity http, OktaOAuth2Properties oktaOAuth2Properties) throws Exception {
152 
153         if (!isEmpty(oktaOAuth2Properties.getClientId()) && !isEmpty(oktaOAuth2Properties.getClientSecret())) {
154             http.oauth2ResourceServer().opaqueToken();
155         }
156     }
157 
158     private OAuth2AccessTokenResponseClient<OAuth2AuthorizationCodeGrantRequest> accessTokenResponseClient(RestTemplate restTemplate) {
159 
160         DefaultAuthorizationCodeTokenResponseClient accessTokenResponseClient = new DefaultAuthorizationCodeTokenResponseClient();
161         accessTokenResponseClient.setRestOperations(restTemplate);
162 
163         return accessTokenResponseClient;
164     }
165 }
